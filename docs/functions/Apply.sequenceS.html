    Example

    import * as E from 'fp-ts/Either'
    import { sequenceS } from 'fp-ts/Apply'

    const ado = sequenceS(E.Apply)

    assert.deepStrictEqual(
      ado({
        a: E.right(1),
        b: E.right(true)
      }),
      E.right({ a: 1, b: true })
    )
    assert.deepStrictEqual(
      ado({
        a: E.right(1),
        b: E.left('error')
      }),
      E.left('error')
    )
